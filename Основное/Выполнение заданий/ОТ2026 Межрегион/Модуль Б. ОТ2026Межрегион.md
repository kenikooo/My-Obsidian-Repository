
`playbook4_k8s_nodes.yaml`
```bash
---
- name: k8s
  hosts: all
  become: yes
  tasks:
	- name: Отключение swap
	  ansible.builtin.command:
	  command: swapoff -a
	  
	- name: Удаление записи из /etc/fstab
	  ansible.builtin.replace:
	    path: /etc/fstab
	    regexp: '^([^#\s]+\s+swap\s+.*)$'
	    replace: '# \1'
	  register: fstab_result
	  
	- name: Загрузка модулей 
	  community.general.modprobe:
	    name: "{{ item }}"
	    state: present
	  loop:
	    - overlay
	    - br_netfilter
	    - nf_conntrack
	  
	- name: Загрузка модулей ядра в файл
	  ansible.builtin.copy:
	    content: |
	      overlay
	      br_netfilter
	      nf_conntrack
	    dest: /etc/modules.load.d/k8s.conf
	    mode: '0644'
	    
	- name: Загрузка модулей ipvs 
	  community.general.modprobe:
	    name: "{{ item }}"
	    state: present
	  loop:
	    - ip_vs
	    - ip_vs_rr
	    - ip_vs_wrr
	    - ip_vs_sh
	    - ip_set
	      	  
	- name: Загрузка модулей ядра в файл ipvs
	  ansible.builtin.copy:
	    content |
	      ip_vs
	      ip_vs_rr
	      ip_vs_wrr
	      ip_vs_sh
	      ip_sh
	    dest: /etc/modules.load.d/ipvs.conf
	    mode: '0644'
	 
     - name: Загрузка модулей ядра в файл k8s.conf
	  ansible.builtin.copy:
	    content |
	      net.bridge.bridge-nf-call-iptables = 1
	      net.bridge.bridge-nf-call-ip6tables = 1
	      net.ipv4.ip_forward = 1
	    dest: /etc/sysctl.d/k8s.conf
	    mode: '0644'
	    
	 - name: Перезагрузка модулей
	   ansible.builtin.command:
		 cmd: sysctl --system
```
